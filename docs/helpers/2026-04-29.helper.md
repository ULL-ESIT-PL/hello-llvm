# slide 5: IR and the compilation process
# llc compiles LLVM IR to assembly -filetype=obj produces an object file
clang -S -emit-llvm examples/factorial.c -o examples/factorial.ll
clang -S -emit-llvm examples/factorial-main.c -o examples/factorial-main.ll 
clang examples/factorial-main.ll examples/factorial.ll -o tmp/f

llvm-link examples/factorial-main.ll examples/factorial.ll -o tmp/combined.bc
llvm-dis < tmp/combined.bc
lli tmp/combined.bc    
llc -filetype=obj tmp/combined.bc -o tmp/combined.o
#### llc does not link libc nor resolve printf, so we need clang to link the final executable
#### Before doing this, ensure both .ll files have compatible target triple and datalayout, otherwise llvm-link may fail or silently produce broken code.

clang tmp/combined.o -o tmp/f
